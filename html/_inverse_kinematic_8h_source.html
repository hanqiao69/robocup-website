56  static bool calcLegJoints(const Pose3D& position, JointData& jointData, bool left, const RobotDimensions& robotDimensions)
57  {
58  Pose3D target(position);
59  JointData::Joint firstJoint(left ? JointData::LHipYawPitch : JointData::RHipYawPitch);
60  const float sign(left ? -1.0f : 1.0f);
61  target.translation.y += (float) robotDimensions.lengthBetweenLegs / 2.f * sign; // translate to origin of leg
62  // rotate by 45° around origin for Nao
63  // calculating sqrt(2) is faster than calculating the resp. rotation matrix with getRotationX()
64  static const float sqrt2_2 = sqrtf(2.0f) * 0.5f;
65  RotationMatrix rotationX_pi_4 = RotationMatrix(Vector3<>(1, 0, 0),
66  Vector3<>(0, sqrt2_2, sqrt2_2 * sign),
67  Vector3<>(0, sqrt2_2 * -sign, sqrt2_2));
68  target.translation = rotationX_pi_4 * target.translation;
69  target.rotation = rotationX_pi_4 * target.rotation;
70 
71  target = target.invert(); // invert pose to get position of body relative to foot
72 
73  // use geometrical solution to compute last three joints
74  float length = target.translation.abs();
75  float sqrLength = length * length;
76  float upperLeg = robotDimensions.upperLegLength;
77  float sqrUpperLeg = upperLeg * upperLeg;
78  float lowerLeg = robotDimensions.lowerLegLength;
79  float sqrLowerLeg = lowerLeg * lowerLeg;
80  float cosLowerLeg = (sqrLowerLeg + sqrLength - sqrUpperLeg) / (2 * lowerLeg * length);
81  float cosKnee = (sqrUpperLeg + sqrLowerLeg - sqrLength) / (2 * upperLeg * lowerLeg);
82 
83  // clip for the case of unreachable position
84  const Range<> clipping(-1.0f, 1.0f);
85  bool reachable = true;
86  if(!clipping.isInside(cosKnee) || clipping.isInside(cosLowerLeg))
87  {
88  cosKnee = clipping.limit(cosKnee);
89  cosLowerLeg = clipping.limit(cosLowerLeg);
90  reachable = false;
91  }
92  float joint3 = pi - acosf(cosKnee); // implicitly solve discrete ambiguousness (knee always moves forward)
93  float joint4 = -acosf(cosLowerLeg);
94  joint4 -= atan2f(target.translation.x, Vector2<>(target.translation.y, target.translation.z).abs());
95  float joint5 = atan2f(target.translation.y, target.translation.z) * sign;
96 
97  // calulate rotation matrix before hip joints
98  RotationMatrix hipFromFoot;
99  hipFromFoot.rotateX(joint5 * -sign);
100  hipFromFoot.rotateY(-joint4 - joint3);
101 
102  // compute rotation matrix for hip from rotation before hip and desired rotation
103  RotationMatrix hip = hipFromFoot.invert() * target.rotation;
104 
105  // compute joints from rotation matrix using theorem of euler angles
106  // see http://www.geometrictools.com/Documentation/EulerAngles.pdf
107  // this is possible because of the known order of joints (z, x, y seen from body resp. y, x, z seen from foot)
108  float joint1 = asinf(-hip[2].y) * -sign;
109  joint1 -= pi_4; // because of the 45°-rotational construction of the Nao legs
110  float joint2 = -atan2f(hip[2].x, hip[2].z);
111  float joint0 = atan2f(hip[0].y, hip[1].y) * -sign;
112 
113  // set computed joints in jointData
114  jointData.angles[firstJoint + 0] = joint0;
115  jointData.angles[firstJoint + 1] = joint1;
116  jointData.angles[firstJoint + 2] = joint2;
117  jointData.angles[firstJoint + 3] = joint3;
118  jointData.angles[firstJoint + 4] = joint4;
119  jointData.angles[firstJoint + 5] = joint5;
120 
121  return reachable;

234  static bool calcElbowPosition(Vector3<> &target, const Vector3<> &targetDir, int side, Vector3<> &elbow, const RobotDimensions& theRobotDimensions, const JointCalibration& theJointCalibration)
235  {
236  const Vector3<> M1(0, 0, 0); //shoulder
237  const Vector3<> M2(target); //hand
238  const float r1 = theRobotDimensions.upperArmLength;
239  const float r2 = theRobotDimensions.lowerArmLength;
240  const Vector3<> M12 = M2 - M1;
241 
242  Vector3<> n = target;
243  n.normalize();
244 
245  //center of intersection circle of spheres around shoulder and hand
246  const Vector3<> M3 = M1 + M12 * ((sqr(r1) - sqr(r2)) / (2 * M12.squareAbs()) + 0.5f);
247 
248  //calculate radius of intersection circle
249  const Vector3<> M23 = M3 - M2;
250  float diff = sqr(r2) - M23.squareAbs();
251  const float radius = sqrtf(diff);
252 
253  //determine a point on the circle
254  const bool specialCase = n.x == 1 && n.y == 0 && n.z == 0 ? true : false;
255  const Vector3<> bla(specialCase ? 0.0f : 1.0f, specialCase ? 1.0f : 0.0f, 0.0f);
256  const Vector3<> pointOnCircle = M3 + (n ^ bla).normalize(radius);
257 
258  //find best point on circle
259  float angleDiff = pi * 2.0f / 3.0f;
260  Vector3<> bestMatch = pointOnCircle;
261  float bestAngle = 0.0f;
262  float newBestAngle = bestAngle;
263  float bestQuality = -2.0f;
264 
265  Vector3<> tDir = targetDir;
266  tDir.normalize();
267 
268  const int offset = side == 1 ? 0 : 4;
269  int iterationCounter = 0;
270  const float maxAngleEpsilon = 1.0f * pi / 180.0f;
271  while(2.0f * angleDiff > maxAngleEpsilon)
272  {
273  for(int i = -1; i <= 1; i++)
274  {
275  if(i == 0 && iterationCounter != 1)
276  continue;
277 
278  iterationCounter++;
279 
280  const Pose3D elbowRotation(RotationMatrix(n, bestAngle + angleDiff * (float)i));
281  const Vector3<> possibleElbow = elbowRotation * pointOnCircle;
282  const Vector3<> elbowDir = (M3 - possibleElbow).normalize();
283  float quality = elbowDir * tDir;
284  if(quality > bestQuality)
285  {
286  bestQuality = quality;
287  bestMatch = possibleElbow;
288  newBestAngle = bestAngle + angleDiff * (float)i;
289  }
290  }
291  angleDiff /= 2.0f;
292  bestAngle = newBestAngle;
293  }
294  //printf("iterations %d\n", iterationCounter);
295  if(bestQuality == -2.0f)
296  return false;
297 
298  //special case of target-out-of-joints-limit problem
299  JointData tAJR;
300  calcJointsForElbowPos(bestMatch, target, tAJR, offset, theRobotDimensions);
301 
302  const JointCalibration::JointInfo ji = theJointCalibration.joints[JointData::LShoulderPitch + offset + 1];
303  if(tAJR.angles[offset + 1] < ji.minAngle)
304  {
305  tAJR.angles[offset + 1] = ji.minAngle;
306  Pose3D shoulder2Elbow;
307  shoulder2Elbow.translate(0, -theRobotDimensions.upperArmLength, 0);
308  shoulder2Elbow.rotateX(-(tAJR.angles[offset + 1] - pi / 2.0f));
309  shoulder2Elbow.rotateY(tAJR.angles[offset + 0] + pi / 2.0f);
310  Vector3<> handInEllbow = shoulder2Elbow * target;
311 
312  handInEllbow.normalize(theRobotDimensions.lowerArmLength);
313  target = shoulder2Elbow.invert() * handInEllbow;
314  bestMatch = shoulder2Elbow.invert() * Vector3<>(0, 0, 0);
315  }
316 
317  elbow = bestMatch;
318  return true;
319  }

321  static void calcJointsForElbowPos(const Vector3<> &elbow, const Vector3<> &target, JointData& targetJointData, int offset, const RobotDimensions& theRobotDimensions)
322  {
323  //set elbow position with the pitch/yaw unit in the shoulder
324  targetJointData.angles[offset + 0] = atan2f(elbow.z, elbow.x);
325  targetJointData.angles[offset + 1] = atan2f(elbow.y, sqrtf(sqr(elbow.x) + sqr(elbow.z)));
326 
327  //calculate desired elbow "klapp"-angle
328  const float c = target.abs(),
329  a = theRobotDimensions.upperArmLength,
330  b = theRobotDimensions.lowerArmLength;
331 
332  //cosine theorem
333  float cosAngle = (-sqr(c) + sqr(b) + sqr(a)) / (2.0f * a * b);
334  if(cosAngle < -1.0f)
335  {
336 // ASSERT(cosAngle > -1.1);
337  cosAngle = -1.0f;
338  }
339  else if(cosAngle > 1.0f)
340  {
341  ASSERT(cosAngle < 1.1);
342  cosAngle = 1.0f;
343  }
344  targetJointData.angles[offset + 3] = acosf(cosAngle) - pi;
345 
346  //calculate hand in elbow coordinate system and calculate last angle
347  Pose3D shoulder2Elbow;
348  shoulder2Elbow.translate(0, -theRobotDimensions.upperArmLength, 0);
349  shoulder2Elbow.rotateX(-(targetJointData.angles[offset + 1] - pi / 2.0f));
350  shoulder2Elbow.rotateY(targetJointData.angles[offset + 0] + pi / 2.0f);
351  const Vector3<> handInEllbow = shoulder2Elbow * target;
352 
353  targetJointData.angles[offset + 2] = -(atan2f(handInEllbow.z, handInEllbow.x) + pi / 2.f);
354  while(targetJointData.angles[offset + 2] > pi)
355  targetJointData.angles[offset + 2] -= 2 * pi;
356  while(targetJointData.angles[offset + 2] < -pi)
357  targetJointData.angles[offset + 2] += 2 * pi;
358  }

368  static void calcArmJoints(const Vector3<>& position, const float elbowYaw, JointData& jointData, bool left, const RobotDimensions& robotDimensions)
369  {
370  JointData::Joint firstJoint(left ? JointData::LShoulderPitch : JointData::RShoulderPitch);
371  const float sign(left ? -1.0f : 1.0f);
372  const Vector3<> pos(position - Vector3<>(robotDimensions.armOffset.x, robotDimensions.armOffset.y * -sign, robotDimensions.armOffset.z));
373  float& joint0 = jointData.angles[firstJoint + 0];
374  float& joint1 = jointData.angles[firstJoint + 1];
375  const float& joint2 = jointData.angles[firstJoint + 2] = elbowYaw;
376  float& joint3 = jointData.angles[firstJoint + 3];
377 
378  // distance of the end effector position to the origin
379  const float positionAbs = pos.abs();
380 
381  // the upper and lower arm form a triangle with the air line to the end effector position being the third edge. Elbow angle can be computed using cosine theorem
382  const float actualUpperArmLength = Vector2<>(robotDimensions.upperArmLength, robotDimensions.yElbowShoulder).abs();
383  float cosElbow = (sqr(actualUpperArmLength) + sqr(robotDimensions.lowerArmLength) - sqr(positionAbs)) / (2.0f * robotDimensions.upperArmLength * robotDimensions.lowerArmLength);
384  // clip for the case of unreachable position
385  cosElbow = Range<>(-1.0f, 1.0f).limit(cosElbow);
386  // elbow is streched in zero-position, hence pi - innerAngle
387  joint3 = -(pi - acosf(cosElbow));
388  // compute temporary end effector position from known third and fourth joint angle
389  const Pose3D tempPose = Pose3D(robotDimensions.upperArmLength, robotDimensions.yElbowShoulder * -sign, 0).rotateX(joint2 * -sign).rotateZ(joint3 * -sign).translate(robotDimensions.lowerArmLength, 0, 0);
390 
391  /* offset caused by third and fourth joint angle */ /* angle needed to realise y-component of target position */
392  joint1 = atan2f(tempPose.translation.y * sign, tempPose.translation.x) + asinf(pos.y / Vector2<>(tempPose.translation.x, tempPose.translation.y).abs()) * -sign;
393  // refresh temporary endeffector position with known joint1
394  const Pose3D tempPose2 = Pose3D().rotateZ(joint1 * -sign).conc(tempPose);
395 
396  /* first compensate offset from temporary position */ /* angle from target x- and z-component of target position */
397  joint0 = -atan2f(tempPose2.translation.z, tempPose2.translation.x) + atan2f(pos.z, pos.x);
398  }
